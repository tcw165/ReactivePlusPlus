50  void defer_at(time_point time_point, constraint::schedulable_fn auto&& fn) const
51  {
52  if (!m_sub.is_subscribed())
53  return;
54 
55  auto& queue = get_schedulable_queue();
56  const bool someone_owns_queue = queue.has_value();
57 
58  const auto drain_on_exit = utils::finally_action(!someone_owns_queue ? &drain_queue : +[]{});
59 
60  if (!someone_owns_queue)
61  {
62  queue = std::priority_queue<current_thread_schedulable>{};
63 
64  // do immediate scheduling till queue is empty
65  while (m_sub.is_subscribed() && get_schedulable_queue()->empty())
66  {
67  std::this_thread::sleep_until(time_point);
68 
69  if (!m_sub.is_subscribed())
70  return;
71 
72  if (const auto duration = fn())
73  time_point = std::max(now(), time_point + duration.value());
74  else
75  return;
76  }
77  }
78 
79  defer_at(time_point, trampoline_schedulable{ *this, time_point, std::forward<decltype(fn)>(fn) });
80  }
81 
82  void defer_at(time_point time_point, trampoline_schedulable&& fn) const
83  {
84  if (!m_sub.is_subscribed())
85  return;
86 
87  get_schedulable_queue()->emplace(time_point, std::move(fn), m_sub);
88  }
89 
90  static time_point now() { return clock_type::now(); }
91 
92  private:

94  };

96  static void drain_queue()
97  {
98  auto& queue = get_schedulable_queue();
99  auto reset_at_final = utils::finally_action{ [] { get_schedulable_queue().reset(); } };
100  std::optional<trampoline_schedulable> function{};
101  while (!queue->empty())
102  {
103  const auto& top = queue->top();
104 
105  wait_and_extract_executable_if_subscribed(top, function);
106 
107  // firstly we need to pop schedulable from queue due to execution of function can add new schedulable
108  queue->pop();
109 
110  if (function)
111  (*function)();
112 
113  function.reset();
114  }
115  }

117  static void wait_and_extract_executable_if_subscribed(const current_thread_schedulable& schedulable, std::optional<trampoline_schedulable>& out)
118  {
119  if (!schedulable.is_subscribed())
120  return;
121 
122  std::this_thread::sleep_until(schedulable.get_time_point());
123 
124  if (!schedulable.is_subscribed())
125  return;
126 
127  out.emplace(std::move(schedulable.extract_function()));
128  }
