26  void emplace(time_point time_point, std::invocable auto&& fn)
27  {
28  emplace_safe(time_point, std::forward<decltype(fn)>(fn));
29  m_cv.notify_one();
30  }

32  bool is_empty() const
33  {
34  std::lock_guard lock{ m_mutex };
35  return m_queue.empty();
36  }
37 
38  bool is_any_ready_schedulable() const
39  {
40  std::lock_guard lock{ m_mutex };
41  return is_any_ready_schedulable_unsafe();
42  }
43 
44  bool pop_if_ready(std::function<void()>& out)
45  {
46  std::lock_guard lock{ m_mutex };
47  if (!is_any_ready_schedulable_unsafe())
48  return false;
49 
50  out = std::move(m_queue.top().ExtractFunction());
51  m_queue.pop();
52  return true;
53  }
54 
55  bool pop_with_wait(std::function<void()>& out, const std::stop_token& token)
56  {
57  while (!token.stop_requested())
58  {
59  std::unique_lock lock{m_mutex};
60 
61  if (!m_cv.wait(lock, token, [&] { return !m_queue.empty(); }))
62  continue;
63 
64  if (!m_cv.wait_until(lock,
65  token,
66  m_queue.top().GetTimePoint(),
67  std::bind_front(&queue_worker_state::is_any_ready_schedulable_unsafe, this)))
68  continue;
69 
70  out = std::move(m_queue.top().ExtractFunction());
71  m_queue.pop();
72  return true;
73  }
74  return false;
75  }

77  void reset()
78  {
79  std::lock_guard lock{ m_mutex };
80  m_queue = std::priority_queue<schedulable>{};
81  }
82 
83 private:
84  void emplace_safe(time_point time_point, std::invocable auto&& fn)
85  {
86  std::lock_guard lock{ m_mutex };
87  m_queue.emplace(time_point, ++m_current_id, std::forward<decltype(fn)>(fn));
88  }
89 
90  bool is_any_ready_schedulable_unsafe() const
91  {
92  return !m_queue.empty() && m_queue.top().GetTimePoint() <= clock_type::now();
93  }
