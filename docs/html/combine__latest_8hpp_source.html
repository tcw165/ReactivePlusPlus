53  // Not copyable nor movable.

57  template<size_t I>
58  void offer_next_value(auto&& value, const auto& subscriber)
59  {
60  std::scoped_lock<std::mutex> lock{m_mutex};
61  std::get<I>(m_values) = std::forward<decltype(value)>(value);
62 
63  // If all the optional(s) have value, dispatch to downstream
64  if (has_values_for_observer())
65  {
66  subscriber.on_next(combine_cached_values());
67  }
68  }
69 
70  void offer_completed(const auto& subscriber)
71  {
72  if (++m_completed_count == m_total_completed_count)
73  {
74  subscriber.on_completed();
75  }
76  }
77 
78  private:
79  RPP_NO_UNIQUE_ADDRESS TCombiner m_combiner;
80 
81  std::mutex m_mutex;
82  std::tuple<std::optional<Type>, std::optional<utils::extract_observable_type_t<TOtherObservable>>...> m_values{};
83  std::atomic_size_t m_completed_count{0};
84 
85  const std::size_t m_total_completed_count = 1 + sizeof...(TOtherObservable);
86 
87  bool has_values_for_observer()
88  {
89  return std::apply([](const auto&... cached_values)
90  {
91  // Variadic expansion used
92  return (cached_values.has_value() && ...);
93  }, m_values);
94  }
95 
96  auto combine_cached_values()
97  {
98  return std::apply([&](const auto&... cached_values)
99  {
100  return m_combiner(cached_values.value()...);
101  }, m_values);
102  }
103  };

112  template<size_t...I>
113  void subscribe_other_observables(std::index_sequence<I...>, // Used in compile time for variadic expansion
114  const auto& subscriber,
115  const auto& coordinator) const
116  {
117  // +1 because the first element in tuple is the current observable, and you want to subscribe to the 'other' observables.
118  // (Use variadic expansion to iterate the observables)
119  (subscribe_observable<I + 1>(std::get<I>(m_other_observables), subscriber, coordinator), ...);
120  }

122  template<size_t I, constraint::observable TObservable>
123  static void subscribe_observable(const TObservable& observable, const auto& subscriber, const auto& coordinator)
124  {
125  using ValueType = utils::extract_observable_type_t<TObservable>;
126  observable.subscribe(create_inner_subscriber<ValueType, I>(subscriber, coordinator));
127  }

129  template<typename ValueType, size_t I>
130  static auto create_inner_subscriber(auto&& subscriber, const auto& coordinator)
131  {
132  auto subscription = subscriber.get_subscription().make_child();
133  return create_subscriber_with_state<ValueType>(
134  std::move(subscription),
135  [coordinator](auto&& value, const auto& subscriber)
136  {
137  coordinator->template offer_next_value<I>(std::forward<decltype(value)>(value), subscriber);
138  },
139  utils::forwarding_on_error{},
140  [coordinator](const auto& subscriber)
141  {
142  coordinator->offer_completed(subscriber);
143  },
144  std::forward<decltype(subscriber)>(subscriber));
145  }
146 
147 public:
148  template<constraint::subscriber_of_type<DownstreamType> TSub>
149  auto operator()(TSub&& subscriber) const
150  {
151  auto coordinator = std::make_shared<combine_latest_coordinator_t>(m_combiner);
152 
153  // Subscribe to other observables and redirect on_next event to coordinator
154  subscribe_other_observables(std::index_sequence_for<TOtherObservable...>{}, subscriber, coordinator);
155 
156  // Redirect values from this observable to the coordinator for value composition
157  return create_inner_subscriber<Type, s_index_of_source_type>(subscriber, coordinator);
158  }
