38 void iterate(auto&& iterable,
39  const schedulers::constraint::scheduler auto& scheduler,
40  constraint::subscriber auto&& subscriber)
41 {
42  if constexpr (constraint::decayed_same_as<decltype(scheduler), schedulers::immediate>)
43  {
44  for (const auto& v : extract_iterable_from_packed(iterable))
45  {
46  if (subscriber.is_subscribed())
47  subscriber.on_next(utils::as_const(v));
48  else
49  return;
50  }
51  subscriber.on_completed();
52  }
53  else
54  {
55  auto worker = scheduler.create_worker(subscriber.get_subscription());
56  worker.schedule([iterable=std::forward<decltype(iterable)>(iterable),
57  subscriber=std::forward<decltype(subscriber)>(subscriber),
58  index = size_t{0}]() mutable-> schedulers::optional_duration
59  {
60  if (!subscriber.is_subscribed())
61  return std::nullopt;
62 
63  try
64  {
65  const auto& extracted_iterable = extract_iterable_from_packed(iterable);
66  const auto end = std::cend(extracted_iterable);
67  auto itr = std::cbegin(extracted_iterable);
68 
69  std::ranges::advance(itr, static_cast<int64_t>(index), end);
70 
71  if (itr != end)
72  {
73  subscriber.on_next(utils::as_const(*itr));
74  if (std::next(itr) != end) // it was not last
75  {
76  ++index;
77  return schedulers::duration{}; // re-schedule this
78  }
79  }
80 
81  subscriber.on_completed();
82  }
83  catch(...)
84  {
85  subscriber.on_error(std::current_exception());
86  }
87  return std::nullopt;
88 
89  });
90  }
91 }
