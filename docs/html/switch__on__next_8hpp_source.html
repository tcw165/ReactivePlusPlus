28 struct switch_on_next_state_t : public std::enable_shared_from_this<switch_on_next_state_t>
29 {
30  auto on_new_observable_switch()
31  {
32  auto state = shared_from_this();
33  auto on_completed = [=](const constraint::subscriber auto& sub)
34  {
35  if (state->count_of_on_completed.load(std::memory_order::acquire) == 1) // 1 because decrement happens in composite_subscription_callback
36  sub.on_completed();
37  };
38 
39  return [=]<constraint::observable TObs>(const TObs& new_observable, const constraint::subscriber auto& sub)
40  {
41  using ValueType = utils::extract_observable_type_t<TObs>;
42 
43  state->current_inner_observable.unsubscribe();
44  state->current_inner_observable = sub.get_subscription().make_child();
45  state->current_inner_observable.add([state, remove_from = sub.get_subscription()]
46  {
47  remove_from.remove(state->current_inner_observable);
48  state->count_of_on_completed.fetch_sub(1, std::memory_order::relaxed);
49  });
50 
51 
52  new_observable.subscribe(combining::create_proxy_subscriber<ValueType>(state->current_inner_observable,
53  sub,
54  state->count_of_on_completed,
55  utils::forwarding_on_next{},
56  utils::forwarding_on_error{},
57  on_completed));
58  };
59  }
60 
61  std::atomic_size_t count_of_on_completed{};
62 private:
63  rpp::composite_subscription current_inner_observable = rpp::composite_subscription::empty();
64 };

74  {
75  const auto state = std::make_shared<switch_on_next_state_t>();
76

78  state->count_of_on_completed,
79  state->on_new_observable_switch(),
80  utils::forwarding_on_error{},
81  [=](const constraint::subscriber auto& sub)
82  {
83  if (state->count_of_on_completed.fetch_sub(1, std::memory_order::acq_rel) == 1)
84  sub.on_completed();
85  });
