19 template<typename T>
20 concept worker_strategy = std::copyable<T> && requires(const T t)
21 {
22  t.defer_at(time_point{}, std::declval<void(*)()>());
23  { t.now() } -> std::same_as<time_point>;
24 };

47  void schedule(time_point time_point, constraint::schedulable_fn auto&& fn) const
48  {
49  m_strategy.defer_at(time_point, schedulable_wrapper<std::decay_t<decltype(fn)>>{m_strategy, time_point, std::forward<decltype(fn)>(fn)});
50  }

69  void operator()()
70  {
71  if (auto duration = m_fn())
72  {
73  if (duration.value() != rpp::schedulers::duration::zero())
74  m_time_point += duration.value();
75  else
76  m_time_point = m_strategy.now();
77 
78  auto time_to_schedule = m_time_point;
79  m_strategy.defer_at(time_to_schedule, std::move(*this));
80  }
